[HEADER]
Title=Exemple de script - Concurrents 
Description=Voici un exemple de script pour Capitalism Lab, listant toutes les variables dans la section [CONCURENTS]. 

[COMPETITORS]
Number of Competitors=20
// Number of Competitors=<1 to 30>
// You may set the number of competitors to any number between 1 and 30. 

Competitor Start-up Capital=High
// Competitor Start-up Capital=<Very Low, Low, Moderate, High, Very High>
 
Competitor Aggressiveness=High
// Competitor Aggressiveness=<Very Low, Low, Moderate, High, Very High>

Competitor Expertise Level=Moderate
// Competitor Expertise Level=<None, Low, Moderate, High>

Show Competitor Trade Secrets=Yes
// Show Competitor Trade Secrets=<Yes, No>

AI Friendly Merger=On
// AI Friendly Merger=<On, Off>

Competence of Local Competitors=High 
// Competence of Local Competitors=<Very Low, Low, Moderate, High, Very High>


// Vous pouvez définir les objectifs commerciaux des sociétés d'IA à l'aide des lignes de script suivantes :
// (Le nombre total de concurrents défini ici ne doit pas dépasser le nombre figurant dans la ligne Nombre de concurrents).

Retail Focused Companies=3
Stock Focused Companies=2
Real Estate Focused Companies=5
Tech Focused Companies=2
Media Focused Companies=2
Diversified Companies=4

AI Product Expertise=Apparel, 3
// Avec la mention ci-dessus, il y aura 3 concurrents AI possédant une expertise dans le domaine de l'habillement au début du jeu. 

// AI Product Expertise=<Expertise Product Class>, <Number of AI competitors> 

// Utilisez ces informations pour attribuer un type d'expertise spécifique aux concurrents de l'IA. 
// <Expertise Product Class> is the expertise product class. For example, ¡°apparel¡±
// <Number of AI competitors> est le nombre de concurrents AI possédant ce type d'expertise. 
// Veuillez noter que cela ne s'applique qu'aux concurrents de l'IA ayant une stratégie commerciale diversifiée. (par exemple, les entreprises axées sur le commerce de détail et les entreprises ayant d'autres centres d'intérêt ne sont pas concernées).
